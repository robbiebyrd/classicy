# ClassicyTriangle Extraction Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extract the disclosure triangle (SVG + click/keyboard toggle + rotation) out of `ClassicyDisclosure` into a standalone, reusable `ClassicyTriangle` component that works fully on its own, while `ClassicyDisclosure` keeps its exact current behavior by using it in a non-interactive, externally-driven mode.

**Architecture:** New component folder `src/SystemFolder/SystemResources/Triangle/` holds `ClassicyTriangle.tsx` (component), `ClassicyTriangle.scss` (styles, renamed from `classicyDisclosureTriangle*` to `classicyTriangle*`), and `ClassicyTriangle.test.tsx`. `ClassicyTriangle` supports controlled (`open`/`onToggle`) and uncontrolled (`defaultOpen`) state, and an `interactive` flag (default `true`) that toggles between a fully self-contained clickable/keyboard-accessible widget and a purely visual SVG. `ClassicyDisclosure` is then updated to render `<ClassicyTriangle interactive={false} />` inside its existing header, dropping its own inline SVG and triangle-specific analytics call, but keeping its header's `role="button"` wrapper as the single click/focus target.

**Tech Stack:** React + TypeScript, classnames, Vitest + Testing Library (`@/__tests__/test-utils`), SCSS (co-located, no Tailwind/inline styles).

## Global Constraints

- Follow the per-component folder convention: `ComponentName.tsx` + `ComponentName.scss` + `ComponentName.test.tsx` co-located in their own folder under `src/SystemFolder/SystemResources/`.
- No manual edits to any `index.ts` barrel file — barrels are auto-generated by `barrelsby` during `pnpm build:source` and are not checked into git.
- SCSS files must `@use '../../ControlPanels/AppearanceManager/styles/appearance';` at the top, matching every existing component stylesheet in this directory.
- Tests mock co-located `.scss` imports via `vi.mock("path/to/File.scss", () => ({}))`, matching `ClassicyDisclosure.test.tsx`.
- `ClassicyDisclosure`'s public props, rendered DOM structure for its header (`role="button"`, `tabIndex`, `onClick`, `onKeyDown` on the header div), and existing test suite (`ClassicyDisclosure.test.tsx`) must remain unchanged and passing without modification.
- Run `pnpm lint` and the relevant Vitest suites before each commit; this repo uses biome (see the existing `biome-ignore lint/a11y/useSemanticElements` comment pattern for div-with-`role="button"` elements).

```tsx
import "./ClassicyTriangle.scss";
import classNames from "classnames";
import {
	type FC as FunctionalComponent,
	type KeyboardEvent,
	useState,
} from "react";
import { useClassicyAnalytics } from "@/SystemFolder/SystemResources/Analytics/useClassicyAnalytics";

export type ClassicyTriangleDirection = "up" | "right" | "down" | "left";

type ClassicyTriangleProps = {
	direction?: ClassicyTriangleDirection;
	open?: boolean;
	defaultOpen?: boolean;
	onToggle?: (open: boolean) => void;
	interactive?: boolean;
	className?: string;
};

export const ClassicyTriangle: FunctionalComponent<ClassicyTriangleProps> = ({
	direction = "right",
	open,
	defaultOpen = false,
	onToggle,
	interactive = true,
	className,
}) => {
	const [internalOpen, setInternalOpen] = useState(defaultOpen);
	const { track } = useClassicyAnalytics();
	const isControlled = open !== undefined;
	const resolvedOpen = isControlled ? open : internalOpen;

	const triangleClassOpenName =
		"classicyTriangle" +
		direction.charAt(0).toUpperCase() +
		direction.slice(1) +
		(resolvedOpen ? "Open" : "Closed");

	function toggle() {
		const next = !resolvedOpen;
		track("click", { expanded: next, type: "ClassicyTriangle", direction });
		if (!isControlled) {
			setInternalOpen(next);
		}
		onToggle?.(next);
	}

	function handleKeyPress(e: KeyboardEvent<HTMLDivElement>) {
		switch (e.key) {
			case "Enter":
			// falls through
			case " ": {
				toggle();
				break;
			}
		}
	}

	const svgElement = (
		<svg
			id="a"
			xmlns="http://www.w3.org/2000/svg"
			viewBox="0 0 6.44 11.12"
			className={classNames(
				"classicyTriangle",
				triangleClassOpenName,
				!interactive && className,
			)}
		>
			<title>Toggle</title>
			<polygon
				className={"classicyTriangleDropShadow"}
				points="6.44 6.05 1.17 1.07 .93 11.12 6.44 6.05"
			/>
			<polygon
				className={"classicyTriangleOutline"}
				points="5.68 5.34 0 0 0 10.68 5.68 5.34"
			/>
			<polygon
				className={"classicyTriangleHighlight"}
				points="4.79 5.34 .76 1.82 .76 8.86 4.79 5.34"
			/>
			<polygon
				className={"classicyTriangleInner"}
				points="4.79 5.34 1.27 3.42 1.29 8.43 4.79 5.34"
			/>
			<polygon
				className={"classicyTriangleShadow"}
				points=".76 8.29 .76 8.86 4.79 5.34 4.47 5.05 .76 8.29"
			/>
		</svg>
	);

	if (!interactive) {
		return svgElement;
	}

	return (
		// biome-ignore lint/a11y/useSemanticElements: triangle wrapper is a block container with svg child incompatible with <button>
		<div
			role="button"
			aria-expanded={resolvedOpen}
			tabIndex={0}
			className={className}
			onClick={toggle}
			onKeyDown={handleKeyPress}
		>
			{svgElement}
		</div>
	);
};
```

```tsx
import {
	ClassicyControlLabel,
	type ClassicyControlLabelSize,
	type ClassicyLabelPosition,
} from "@/SystemFolder/SystemResources/ControlLabel/ClassicyControlLabel";
import { ClassicyTriangle } from "@/SystemFolder/SystemResources/Triangle/ClassicyTriangle";
import "./ClassicyDisclosure.scss";
import classNames from "classnames";
import {
	type FC as FunctionalComponent,
	type KeyboardEvent,
	type ReactNode,
	useState,
} from "react";
import { useClassicyAnalytics } from "@/SystemFolder/SystemResources/Analytics/useClassicyAnalytics";

type ClassicyDisclosureTriangleDirections = "up" | "right" | "down" | "left";

type ClassicyDisclosureProps = {
	direction?: ClassicyDisclosureTriangleDirections;
	label?: string;
	labelSize?: ClassicyControlLabelSize;
	labelPosition?: ClassicyLabelPosition;
	children?: ReactNode;
};

export const ClassicyDisclosure: FunctionalComponent<
	ClassicyDisclosureProps
> = ({
	direction = "right",
	label = "",
	labelSize = "medium",
	labelPosition = "right",
	children,
}) => {
	const [open, setOpen] = useState(false);
	const { track } = useClassicyAnalytics();
	const analyticsArgs = { type: "ClassicyLabel", label };

	function handleKeyPress(e: KeyboardEvent<HTMLDivElement>) {
		switch (e.key) {
			case "Enter":
			// falls through
			case " ": {
				track("click", { expanded: !open, ...analyticsArgs });
				setOpen(!open);
				break;
			}
		}
	}

	return (
		<div className={classNames("classicyDisclosure")}>
			{/* biome-ignore lint/a11y/useSemanticElements: disclosure header is a block container with svg child incompatible with <button> */}
			<div
				role="button"
				className={"classicyDisclosureHeader"}
				onClick={() => {
					track("click", { expanded: !open, ...analyticsArgs });
					setOpen(!open);
				}}
				tabIndex={0}
				onKeyDown={(e) => handleKeyPress(e)}
			>
				{labelPosition === "left" && (
					<ClassicyControlLabel label={label} labelSize={labelSize} />
				)}
				<ClassicyTriangle
					direction={direction}
					open={open}
					interactive={false}
				/>
				{labelPosition !== "left" && (
					<ClassicyControlLabel label={label} labelSize={labelSize} />
				)}
			</div>
			<div
				className={classNames(
					"classicyDisclosureInner",
					open ? "classicyDisclosureInnerOpen" : "classicyDisclosureInnerClose",
				)}
			>
				{children}
			</div>
		</div>
	);
};
```
